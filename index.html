import {inject} from 'aurelia-framework';
import {UserService} from 'users/user-service';
import {EventAggregator} from
	'aurelia-event-aggregator';
import * as toastr from 'toastr';

@inject(UserService, EventAggregator, toastr)
export class Users {

	constructor(userService, eventAggregator, toastr) {
		this.userService = userService;
		this.eventAggregator = eventAggregator;
		this.toastr = toastr;
	}

	activate(params) {
		return this.userService.getAll()
			.then( users => this.users = users );
	}
}

Data Binding: Aurelia

  • Append .bind to any html or custom attribute
  • Defaults to two-way for input, one-way for everthing else
  • Want more control? .one-way, .two-way, .one-time
  • Adaptive Binding used to pick optimal observation strategy and minimize dirty checking

Element will not update because event happened outside of Angular’s context.


link: function(scope, element) {
	// Initialize datepicker
	$('input').datepicker();

	// Update data binding
	$('input').datepicker().on('changeDate',
		function(e) {
			scope.$apply(function () {
				scope.data = e.date;
			});
	});
}

Run update inside scope.$apply to trigger a digest cycle.

Directive: Watch


link: function(scope, element) {
	scope.$watch('data',
		function(newVal) {
    	$('input').datepicker('setDate', newVal);
  	});
}

To populate datepicker from model, use scope.$watch to add to list of watchers evaluated in digest cycle.

Custom Element: Datepicker


bind() {
	// Find input element
	this.selector = $(this.element).find('.date');

	// Initialize datepicker
	this.selector.datepicker();

	// Update bound data on selection
	this.selector.datepicker().on('changeDate', (e) => {
		this.data = e.date;
	});

	// Prepopulate with bound data
	if (this.data) {
		this.selector.datepicker('setDate', this.data);
	}
}

Bind lifecycle method invoked when view is bound.

Changing data in datepicker event just works.

No internal frameowrk knowledge required.
